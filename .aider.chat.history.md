const { ethers } = require("hardhat");

async function main() {
  console.log("Deploying contracts to Base...");
  
  // Deploy SkillNFT

  // Deploy ReputationSystem
  const ReputationSystem = await ethers.getContractFactory("ReputationSystem");
  const reputation = await ReputationSystem.deploy();
  await reputation.deployed();
  console.log("ReputationSystem deployed to:", reputation.address);
  
  // Deploy JobBounty (pass addresses of other contracts)
  const JobBounty = await ethers.getContractFactory("JobBounty");
  const jobBounty = await JobBounty.deploy(
    skillNFT.address,

  // Grant permissions
  await reputation.grantRole(
    await reputation.UPDATER_ROLE(),

  console.log("\nDeployment complete!");

}
